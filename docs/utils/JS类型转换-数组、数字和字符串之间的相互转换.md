### 数组

```js

//数组去逗号，数组转字符串
arr.join('')
arr + ''

//数组转数字
//数组中的字符串元素转数字
var b = ['1','2','3']
b.map(Number) => [1, 2, 3]

JSON.parse('[' + String(['1', '2', '3']) + ']') => [1, 2, 3]

//数组截取
arr.slice(start,end)
```

### 字符串

```js
//字符串转数组
string.split()
var c = '123'
c.split() => ['123']
c.split('') => ['1','2','3']

//字符串转数字
+'123' => 123
Number('123') => 123
JSON.parse('123') => 123
```

### 数字

```js
//数字转字符串
 123 + '' => '123'

//数字转数组
Array.of(1) => [1]

Array.of(1,2,3) => [1,2,3]

```
